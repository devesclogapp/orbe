249:-- FUNÇÃO: get_next_cnab_sequencial
250:-- Retorna próximo sequencial de forma atômica (sem race condition)
251:-- ============================================================
252:CREATE OR REPLACE FUNCTION public.get_next_cnab_sequencial(
253:    p_tenant_id UUID,
254:    p_conta_bancaria_id UUID,
255:    p_banco_codigo TEXT
256:)
257:RETURNS INTEGER
258:LANGUAGE plpgsql
259:SECURITY DEFINER
260:AS $$
261:DECLARE
262:    v_next INTEGER;
263:BEGIN
264:    INSERT INTO public.cnab_sequencial_controle (tenant_id, conta_bancaria_id, banco_codigo, ultimo_sequencial)
265:    VALUES (p_tenant_id, p_conta_bancaria_id, p_banco_codigo, 1)
266:    ON CONFLICT (tenant_id, conta_bancaria_id)
267:    DO UPDATE SET
268:        ultimo_sequencial = cnab_sequencial_controle.ultimo_sequencial + 1,
269:        updated_at = timezone('utc', now())
270:    RETURNING ultimo_sequencial INTO v_next;
272:    RETURN v_next;
273:END;
274:$$;

934:CREATE OR REPLACE FUNCTION public.guard_cnab_retorno_item_conciliacao()
935:RETURNS TRIGGER AS $$
936:DECLARE
937:  v_is_allowed BOOLEAN;
938:BEGIN
939:  v_is_allowed := public.is_financeiro_conciliacao_role();
941:  IF NEW.status IS DISTINCT FROM OLD.status THEN
942:    RAISE EXCEPTION 'O status bancario do retorno nao pode ser alterado manualmente.';
943:  END IF;
945:  IF (

948:    NEW.conciliado_em IS DISTINCT FROM OLD.conciliado_em OR

950:    NEW.revertido_em IS DISTINCT FROM OLD.revertido_em OR
951:    NEW.revertido_por IS DISTINCT FROM OLD.revertido_por
952:  ) AND NOT v_is_allowed THEN
953:    RAISE EXCEPTION 'Usuario sem permissao para alterar a conciliacao.';
954:  END IF;
956:  IF OLD.status_conciliacao = 'conciliado' AND NEW.status_conciliacao NOT IN ('conciliado', 'revertido') THEN
957:    RAISE EXCEPTION 'Item conciliado nao pode ser alterado sem reversao autorizada.';
958:  END IF;
960:  IF OLD.status_conciliacao = 'conciliado' AND (
961:    NEW.valor_esperado IS DISTINCT FROM OLD.valor_esperado OR
962:    NEW.valor_retornado IS DISTINCT FROM OLD.valor_retornado
963:  ) THEN
964:    RAISE EXCEPTION 'Valores de item conciliado nao podem ser alterados silenciosamente.';
965:  END IF;
967:  RETURN NEW;
968:END;
